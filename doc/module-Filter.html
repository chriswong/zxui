Filter

条件过滤器
Source:
  • Filter.js, line 52

Example

var map = {
    '1': [2, 3, 5],
    '2': [1, 2, 4]
}

new Filter({
  prefix: 'ecl-ui-filter',
  main: me.qq('ecl-ui-filter'),
  groups: 'p',
  
  onChange: function (e) {
     
    // load data

    // disable or enable items
    if (e.key === 'type') {
        var value = e.value[0];

        if (map[value]) {
            this.disableItems('special', map[value]);
        }
        else {
            this.enableItems('special');
        }
    }
  }

}).render();
