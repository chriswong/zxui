Select

下拉选择菜单
Source:
  • Select.js, line 84

Example

 new Select({
    prefix: 'ecl-ui-sel',
    main: me.qq('ecl-ui-sel'),
    target: me.qq('ecl-ui-sel-holder'),
    offset: {
      y: -1
    },
    onChange: function (arg) {
      window.console && console.log(arg);
    }
  }).render();
