Select

Source:

Example

 new Select({
    prefix: 'ecl-ui-sel',
    main: me.qq('ecl-ui-sel'),
    target: me.qq('ecl-ui-sel-holder'),
    offset: {
      y: -1
    },
    onChange: function (arg) {
      console.log(arg);
    }
  }).render();
